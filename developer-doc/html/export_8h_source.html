140  bool solve()
141  {
142 
143  // Retrieve the name
144  std::string ptsFileName = exportName + ".pts";
145  std::string pxlFileName = exportName + ".pxl";
146 
147 
148  // Create the files
149  std::ofstream ptsFile;
150  ptsFile.open(ptsFileName);
151  std::ofstream pxlFile;
152  pxlFile.open(pxlFileName);
153 
154  // Get the dependency... has only one.
155  TriangulateWorkplane * dependency = static_cast<TriangulateWorkplane *>(getDependencyRef(0));
156 
157  // Write down all polygons in triangulation
158  std::vector<Triangulation * > * triangulations = &(dependency->triangulations);
159  size_t nP = triangulations->size();
160 
161  // Iterate
162  for (size_t j = 0; j < nP; j++) {
163 
164  // Get the triangulated polygon
165  Triangulation * triangulatedPolygon = triangulations->at(j);
166 
167  // Retrieve the normal
168  Vector3D normal = triangulatedPolygon->getPolygon()->getNormal();
169  double nx = normal.getX();
170  double ny = normal.getY();
171  double nz = normal.getZ();
172 
173  // Count triangles
174  size_t nTriangles = triangulatedPolygon->getNumTriangles();
175 
176  for (size_t k = 0; k < nTriangles; k++) {
177  Triangle * triangle = triangulatedPolygon->getTriangleRef(k);
178 
179  if (triangle == NULL)
180  continue;
181 
182  double x = 0;
183  double y = 0;
184  double z = 0;
185  for (int l = 0; l < 3; l++) {
186  Point3D * p = triangle->getVertex(l);
187  double px = p->getX();
188  double py = p->getY();
189  double pz = p->getZ();
190  x += px;
191  y += py;
192  z += pz;
193  pxlFile << px << GLARE_TAB << py << GLARE_TAB << pz << GLARE_TAB;
194  }
195  pxlFile << "\n";
196  ptsFile << x / 3 << GLARE_TAB << y / 3 << GLARE_TAB << z / 3 << GLARE_TAB << nx << GLARE_TAB << ny << GLARE_TAB << nz << "\n";
197 
198  }
199 
200  }
201 
202 
203  // Close the files
204  ptsFile.close();
205  pxlFile.close();
206 
207 
208  return true;
209  }
